# Справочник конфигураций

Конфиги в этом файле приведены как рабочие шаблоны. Комментарии описывают назначение параметра, возможные значения и влияние изменения значения на поведение Kafka или приложения.

## 1. Standalone Kafka broker в KRaft

Файл: `/opt/kafka/config/kraft/server.properties`

```properties
# node.id
# Возможные значения: целое число от 1 и выше, уникальное внутри KRaft-кластера.
# Влияет на идентификацию broker/controller в metadata quorum и в размещении replicas.
# Нельзя запускать два узла одного кластера с одинаковым node.id: появятся конфликты регистрации и quorum не сможет корректно работать.
node.id=1

# process.roles
# Возможные значения: broker, controller, broker,controller.
# broker обслуживает client requests и хранит partitions.
# controller управляет metadata: brokers, topics, partitions, leader election.
# broker,controller уменьшает число процессов; разделение ролей снижает влияние клиентской нагрузки на controller quorum.
process.roles=broker,controller

# controller.quorum.voters
# Возможные значения: список nodeId@host:port через запятую, например 1@host1:9093,2@host2:9093,3@host3:9093.
# Определяет состав controller quorum в KRaft.
# Для отказоустойчивости используется нечетное число controller voters: 1, 3, 5.
# Кластер из 3 controllers выдерживает отказ 1 controller; из 5 controllers выдерживает отказ 2 controllers.
controller.quorum.voters=1@localhost:9093

# listeners
# Возможные значения: список listenerName://host:port.
# host можно оставить пустым, например PLAINTEXT://:9092, тогда Kafka слушает все интерфейсы.
# Определяет локальные sockets, на которых процесс принимает подключения.
# Ошибка в портах приводит к Address already in use или недоступности broker/controller.
listeners=PLAINTEXT://:9092,CONTROLLER://:9093

# advertised.listeners
# Возможные значения: список listenerName://host:port, доступный клиентам и другим brokers.
# Это адрес, который Kafka возвращает клиентам в metadata response.
# Если указать localhost на удаленном сервере, внешние клиенты получат localhost и не смогут подключиться.
# В Docker, NAT и облаках этот параметр обычно отличается от listeners.
advertised.listeners=PLAINTEXT://localhost:9092

# controller.listener.names
# Возможные значения: одно или несколько имен listeners, например CONTROLLER.
# Указывает, какие listeners используются controller quorum.
# Клиентские приложения не должны подключаться к controller listener.
controller.listener.names=CONTROLLER

# listener.security.protocol.map
# Возможные значения протоколов: PLAINTEXT, SSL, SASL_PLAINTEXT, SASL_SSL.
# Сопоставляет имя listener с протоколом безопасности.
# Если listener объявлен в listeners, но отсутствует в этой карте, broker не сможет стартовать.
# Выбор SSL/SASL_SSL включает требования к TLS/SASL параметрам.
listener.security.protocol.map=PLAINTEXT:PLAINTEXT,CONTROLLER:PLAINTEXT

# inter.broker.listener.name
# Возможные значения: имя listener из listeners и listener.security.protocol.map, например PLAINTEXT, SSL, INTERNAL.
# Определяет listener для трафика broker-to-broker: replication fetch, metadata requests, internal communication.
# При выборе SSL или SASL_SSL межброкерный обмен требует корректных сертификатов или SASL credentials.
inter.broker.listener.name=PLAINTEXT

# log.dirs
# Возможные значения: один или несколько каталогов через запятую.
# Хранит partition logs, indexes и KRaft metadata log, если процесс содержит controller.
# Несколько каталогов позволяют распределить partitions по дискам.
# Потеря каталога без replication factor > 1 приводит к потере данных соответствующих replicas.
log.dirs=/var/lib/kafka/kraft-combined-logs

# num.partitions
# Возможные значения: целое число от 1 и выше.
# Количество partitions по умолчанию для topics, если --partitions не указан при создании.
# Большее значение повышает потенциальный параллелизм producer/consumer, но увеличивает число файлов, metadata и нагрузку на controller.
num.partitions=1

# default.replication.factor
# Возможные значения: целое число от 1 до количества brokers.
# Replication factor по умолчанию для автоматически создаваемых topics.
# В production обычно 3; на single-broker стенде только 1.
# Значение больше числа доступных brokers приведет к ошибке создания topic.
default.replication.factor=1

# offsets.topic.replication.factor
# Возможные значения: 1 на single-broker, 3 или больше в production.
# Replication factor внутреннего topic __consumer_offsets.
# Если значение больше числа brokers, первая consumer group не сможет создать offsets topic.
# Низкое значение снижает отказоустойчивость committed offsets.
offsets.topic.replication.factor=1

# transaction.state.log.replication.factor
# Возможные значения: 1 на single-broker, 3 или больше в production.
# Replication factor внутреннего transaction state log для transactional producers.
# При недостаточном числе brokers transactional producer не сможет работать.
transaction.state.log.replication.factor=1

# transaction.state.log.min.isr
# Возможные значения: 1 на single-broker, 2 при replication factor 3.
# Минимальное число in-sync replicas для transaction state log.
# Чем выше значение, тем выше защита от потери transactional state, но ниже доступность при отказах brokers.
transaction.state.log.min.isr=1

# min.insync.replicas
# Возможные значения: от 1 до replication factor topic.
# Минимум ISR replicas, которые должны подтвердить запись при producer acks=all.
# При replication factor 3 обычно используется 2.
# Если доступных ISR меньше значения, producer получает NOT_ENOUGH_REPLICAS и запись отклоняется.
min.insync.replicas=1

# auto.create.topics.enable
# Возможные значения: true, false.
# true разрешает Kafka создавать topic автоматически при первом обращении producer/consumer.
# false требует явного создания topics и уменьшает риск опечаток в именах topics.
auto.create.topics.enable=false

# log.retention.hours
# Возможные значения: целое число часов; также можно использовать log.retention.ms или log.retention.minutes.
# Определяет срок хранения сообщений для topics с cleanup.policy=delete, если topic-level retention не задан.
# Уменьшение значения ускоряет удаление старых closed segments; active segment не удаляется до закрытия.
log.retention.hours=168

# log.retention.bytes
# Возможные значения: -1 для отключения ограничения по размеру или положительное число байт.
# Ограничивает размер partition log.
# При достижении лимита Kafka удаляет старые closed segments.
# Слишком малое значение может удалить данные раньше, чем consumers успеют их прочитать.
log.retention.bytes=-1

# log.segment.bytes
# Возможные значения: положительное число байт.
# Размер segment file, после которого Kafka открывает новый segment.
# Меньшие segments быстрее подпадают под retention и compaction, но увеличивают число файлов и операций с индексами.
log.segment.bytes=1073741824

# log.retention.check.interval.ms
# Возможные значения: положительное число миллисекунд.
# Частота фоновой проверки segments на удаление по retention.
# Меньшее значение ускоряет реакцию на retention, но добавляет фоновые проверки.
log.retention.check.interval.ms=300000

# log.cleaner.enable
# Возможные значения: true, false.
# true включает cleaner threads для compacted topics.
# false отключает compaction даже для topics с cleanup.policy=compact.
# В production обычно true.
log.cleaner.enable=true
```

## 2. Multi-broker KRaft cluster

Шаблон для broker 1 из трех. Для broker 2 и 3 меняются `node.id`, `listeners`, `advertised.listeners` и `log.dirs`.

```properties
# node.id
# Возможные значения: 1, 2, 3 в кластере из трех nodes.
# Значение должно совпадать с node id в controller.quorum.voters.
# Используется controller quorum, replica assignment и broker registration.
node.id=1

# process.roles
# Возможные значения: broker,controller для combined mode; broker или controller для separated mode.
# combined mode проще: три процесса одновременно хранят данные и metadata.
# separated mode устойчивее под нагрузкой: controller nodes не обслуживают client traffic.
process.roles=broker,controller

# controller.quorum.voters
# Возможные значения: все controller nodes кластера в формате nodeId@host:port.
# Список должен быть одинаковым на всех nodes.
# Для трех voters quorum требует две доступные controller-ноды.
controller.quorum.voters=1@localhost:19093,2@localhost:29093,3@localhost:39093

# listeners
# Возможные значения: любые свободные порты; имена listener должны быть согласованы с listener.security.protocol.map.
# PLAINTEXT принимает producer/consumer/admin clients.
# CONTROLLER обслуживает только controller quorum traffic.
listeners=PLAINTEXT://:19092,CONTROLLER://:19093

# advertised.listeners
# Возможные значения: адреса, доступные клиентам и другим brokers.
# В многоузловом стенде нельзя указывать один и тот же advertised host:port для разных brokers.
# Клиент получает все advertised.listeners через metadata и подключается напрямую к leader partitions.
advertised.listeners=PLAINTEXT://localhost:19092

# controller.listener.names
# Возможные значения: имя listener с protocol PLAINTEXT или SSL, обычно CONTROLLER.
# Этот listener не используется для produce/fetch запросов.
controller.listener.names=CONTROLLER

# listener.security.protocol.map
# Возможные значения: PLAINTEXT, SSL, SASL_PLAINTEXT, SASL_SSL для каждого listener name.
# В production обычно разделяют INTERNAL:SASL_SSL, EXTERNAL:SASL_SSL, CONTROLLER:SSL.
listener.security.protocol.map=PLAINTEXT:PLAINTEXT,CONTROLLER:PLAINTEXT

# inter.broker.listener.name
# Возможные значения: listener для внутреннего трафика, например PLAINTEXT или INTERNAL.
# Если listener недоступен другим brokers, replication остановится и ISR начнет уменьшаться.
inter.broker.listener.name=PLAINTEXT

# log.dirs
# Возможные значения: отдельный каталог на каждый broker.
# Использование одного каталога двумя brokers запрещено: данные и locks будут конфликтовать.
log.dirs=/var/lib/kafka/broker-1

# num.partitions
# Возможные значения: 1 и выше.
# Default partitions для новых topics.
# Значение 3 дает по одной partition на broker при трех partitions и трех brokers.
num.partitions=3

# default.replication.factor
# Возможные значения: от 1 до числа brokers.
# Для кластера из трех brokers значение 3 дает по три copies каждой partition.
# Значение 2 выдерживает потерю одного broker только для части сценариев и снижает запас ISR.
default.replication.factor=3

# min.insync.replicas
# Возможные значения: 1..replication.factor.
# При replication factor 3 значение 2 позволяет пережить отказ одного broker при acks=all.
# Значение 3 требует доступности всех replicas и снижает доступность записи при любом отказе.
min.insync.replicas=2

# offsets.topic.replication.factor
# Возможные значения: 3 для отказоустойчивых consumer offsets.
# Если __consumer_offsets потеряет replicas, consumers могут потерять committed progress.
offsets.topic.replication.factor=3

# transaction.state.log.replication.factor
# Возможные значения: 3 для production-кластера из трех brokers.
# Защищает состояние транзакционных producers и exactly-once flows.
transaction.state.log.replication.factor=3

# transaction.state.log.min.isr
# Возможные значения: обычно 2 при replication factor 3.
# Если ISR transaction log меньше 2, transactional writes будут отклоняться.
transaction.state.log.min.isr=2
```

## 3. Topic-level configuration

Параметры передаются через `kafka-topics.sh --config key=value` или изменяются через `kafka-configs.sh`.

```properties
# cleanup.policy
# Возможные значения: delete, compact, delete,compact.
# delete удаляет старые closed segments по retention.ms или retention.bytes.
# compact оставляет последнее значение для каждого key.
# delete,compact одновременно применяет compaction и удаление по retention.
cleanup.policy=delete

# retention.ms
# Возможные значения: -1 для бессрочного хранения или положительное число миллисекунд.
# Работает для cleanup.policy=delete и delete,compact.
# Уменьшение значения снижает объем хранения, но повышает риск, что медленный consumer не успеет прочитать данные.
retention.ms=604800000

# retention.bytes
# Возможные значения: -1 без ограничения или положительное число байт на partition.
# Ограничивает размер каждой partition, а не всего topic.
# Topic с 10 partitions и retention.bytes=1GB может занять до 10GB до учета replicas.
retention.bytes=-1

# segment.bytes
# Возможные значения: положительное число байт.
# Размер segment file внутри partition.
# Малое значение ускоряет закрытие segments и применение retention/compaction.
# Слишком малое значение в production увеличивает число файлов и нагрузку на filesystem.
segment.bytes=1073741824

# segment.ms
# Возможные значения: положительное число миллисекунд.
# Kafka закрывает active segment по времени, даже если segment.bytes не достигнут.
# Ускоряет применение retention/compaction к старым данным, потому что active segment не очищается.
segment.ms=604800000

# min.cleanable.dirty.ratio
# Возможные значения: число от 0.0 до 1.0.
# Доля "грязных" данных в compacted log, после которой cleaner считает segment кандидатом на compaction.
# Меньшее значение запускает compaction чаще и повышает IO-нагрузку.
min.cleanable.dirty.ratio=0.5

# delete.retention.ms
# Возможные значения: положительное число миллисекунд.
# В compacted topics задает срок хранения tombstone records с value=null.
# Если consumer восстанавливает state из compacted topic медленнее этого срока, он может не увидеть удаление key.
delete.retention.ms=86400000

# max.message.bytes
# Возможные значения: положительное число байт.
# Максимальный размер одного record batch на topic.
# Должен быть согласован с producer max.request.size и broker message.max.bytes.
# Большие сообщения ухудшают latency, memory pressure и recovery.
max.message.bytes=1048588

# compression.type
# Возможные значения: producer, uncompressed, gzip, snappy, lz4, zstd.
# producer сохраняет тип сжатия, выбранный producer.
# lz4 и zstd обычно дают хороший баланс throughput и CPU.
# gzip сильнее сжимает, но потребляет больше CPU.
compression.type=producer

# message.timestamp.type
# Возможные значения: CreateTime, LogAppendTime.
# CreateTime использует timestamp от producer.
# LogAppendTime заменяет timestamp временем записи broker.
# Влияет на time-based retention, windowing и анализ задержек.
message.timestamp.type=CreateTime

# unclean.leader.election.enable
# Возможные значения: true, false.
# false запрещает выбирать out-of-sync replica leader и защищает от потери acknowledged сообщений.
# true повышает доступность чтения/записи при потере ISR, но может привести к потере данных.
unclean.leader.election.enable=false
```

## 4. Producer properties

Файл: `producer.properties` или словарь Python `confluent_kafka.Producer`.

```properties
# bootstrap.servers
# Возможные значения: host:port или список host:port через запятую.
# Используется только для первичного подключения и получения cluster metadata.
# Не обязан содержать все brokers, но в production указывают несколько адресов для отказоустойчивого старта клиента.
bootstrap.servers=localhost:9092

# client.id
# Возможные значения: произвольная строка.
# Попадает в broker logs, metrics и quotas.
# Помогает отличать приложения и экземпляры producer.
client.id=orders-api-1

# acks
# Возможные значения: 0, 1, all или -1.
# 0 не ждет подтверждения broker: максимальный throughput, нет гарантии доставки.
# 1 ждет leader: баланс latency и надежности, но acknowledged запись может потеряться при отказе leader до replication.
# all/-1 ждет все ISR и вместе с min.insync.replicas дает устойчивую запись.
acks=all

# enable.idempotence
# Возможные значения: true, false.
# true включает producer idempotence и защищает от дублей при retries в рамках producer session.
# Требует acks=all, retries>0 и совместимых in-flight settings.
# Используется как основа exactly-once transactional flows.
enable.idempotence=true

# retries
# Возможные значения: 0 и выше.
# Количество повторов при retriable errors.
# Большое значение повышает вероятность успешной доставки при временных сбоях.
# Без идемпотентности retries могут создавать дубли при сетевых неопределенностях.
retries=10

# retry.backoff.ms
# Возможные значения: положительное число миллисекунд.
# Задержка между повторными попытками.
# Увеличение снижает давление на broker при аварии, но увеличивает latency восстановления.
retry.backoff.ms=100

# delivery.timeout.ms
# Возможные значения: положительное число миллисекунд.
# Верхняя граница времени доставки record с учетом batching, retries и ожидания ack.
# Если истекло, producer возвращает timeout even if broker later writes the record.
delivery.timeout.ms=120000

# request.timeout.ms
# Возможные значения: положительное число миллисекунд.
# Сколько producer ждет ответ broker на request.
# Должен быть меньше delivery.timeout.ms.
# Слишком маленькое значение вызывает лишние retries при нормальной latency.
request.timeout.ms=30000

# batch.size
# Возможные значения: положительное число байт.
# Максимальный размер batch на partition.
# Большие batches повышают throughput и compression ratio, но могут увеличить latency и memory usage.
batch.size=32768

# linger.ms
# Возможные значения: 0 и выше.
# Сколько producer ждет дополнительные records перед отправкой неполного batch.
# 0 минимизирует latency; 5-20 ms часто повышают throughput.
linger.ms=10

# compression.type
# Возможные значения: none, gzip, snappy, lz4, zstd.
# Сжимает batches на стороне producer.
# Уменьшает network/disk usage, добавляет CPU cost.
# zstd обычно лучше сжимает, lz4 часто быстрее.
compression.type=lz4

# max.in.flight.requests.per.connection
# Возможные значения: 1 и выше.
# Количество неподтвержденных requests на TCP connection.
# Большое значение повышает throughput.
# Для строгого порядка с retries без идемпотентности используют 1; при idempotence Kafka сохраняет порядок при допустимых настройках.
max.in.flight.requests.per.connection=5

# key.serializer
# Возможные значения: Java serializer class.
# Для console/Java clients определяет преобразование key в bytes.
# В Python confluent-kafka key передается как bytes или string и этот параметр не используется.
key.serializer=org.apache.kafka.common.serialization.StringSerializer

# value.serializer
# Возможные значения: Java serializer class.
# Для JSON без Schema Registry часто используется StringSerializer.
# Для Avro/Protobuf используются serializers с Schema Registry.
value.serializer=org.apache.kafka.common.serialization.StringSerializer
```

## 5. Consumer properties

```properties
# bootstrap.servers
# Возможные значения: host:port или список host:port через запятую.
# Используется для подключения к cluster metadata.
# После metadata consumer подключается к brokers, которые являются leaders нужных partitions.
bootstrap.servers=localhost:9092

# group.id
# Возможные значения: непустая строка.
# Consumers с одинаковым group.id делят partitions между собой.
# Consumers с разными group.id читают один topic независимо и имеют отдельные committed offsets.
group.id=orders-service

# client.id
# Возможные значения: произвольная строка.
# Видна в logs, metrics, quotas.
# Удобна для различения instances внутри одной group.
client.id=orders-service-1

# enable.auto.commit
# Возможные значения: true, false.
# true фиксирует offsets автоматически по interval.
# false требует ручного commit после обработки.
# Для at-least-once обработки с внешней БД обычно false.
enable.auto.commit=false

# auto.commit.interval.ms
# Возможные значения: положительное число миллисекунд.
# Используется только при enable.auto.commit=true.
# Больший interval увеличивает возможный повтор обработки после сбоя.
# Меньший interval добавляет нагрузку на __consumer_offsets.
auto.commit.interval.ms=5000

# auto.offset.reset
# Возможные значения: earliest, latest, none.
# earliest начинает с начала partition, если committed offset отсутствует.
# latest начинает только с новых сообщений.
# none вызывает ошибку, если offset отсутствует; удобно для строгих production-процессов.
auto.offset.reset=earliest

# max.poll.records
# Возможные значения: положительное целое число.
# Максимум records за один poll.
# Большие batches повышают throughput обработки, но увеличивают время до commit и риск rebalance из-за долгой обработки.
max.poll.records=100

# max.poll.interval.ms
# Возможные значения: положительное число миллисекунд.
# Максимальное время между poll вызовами.
# Если обработка длится дольше, broker исключает consumer из group и запускает rebalance.
# Для долгой обработки увеличивают значение или уменьшают max.poll.records.
max.poll.interval.ms=300000

# session.timeout.ms
# Возможные значения: положительное число миллисекунд в пределах broker group.min/group.max.session.timeout.ms.
# Время, после которого consumer считается умершим без heartbeats.
# Меньшее значение быстрее обнаруживает сбой, но чувствительнее к паузам GC/CPU.
session.timeout.ms=45000

# heartbeat.interval.ms
# Возможные значения: положительное число миллисекунд, обычно около session.timeout.ms / 3.
# Частота heartbeats coordinator.
# Слишком большое значение замедляет обнаружение проблем; слишком маленькое увеличивает служебный трафик.
heartbeat.interval.ms=3000

# isolation.level
# Возможные значения: read_uncommitted, read_committed.
# read_uncommitted читает все records, включая aborted transactional writes.
# read_committed скрывает aborted transactional records и нужен для exactly-once transactional pipelines.
isolation.level=read_uncommitted

# partition.assignment.strategy
# Возможные значения: RangeAssignor, RoundRobinAssignor, StickyAssignor, CooperativeStickyAssignor и их class names.
# Определяет распределение partitions между consumers внутри group.
# CooperativeStickyAssignor уменьшает паузы при rebalance за счет incremental cooperative rebalance.
partition.assignment.strategy=org.apache.kafka.clients.consumer.CooperativeStickyAssignor

# key.deserializer
# Возможные значения: Java deserializer class.
# Для console/Java clients преобразует key bytes в объект.
# В Python confluent-kafka приложение декодирует bytes самостоятельно.
key.deserializer=org.apache.kafka.common.serialization.StringDeserializer

# value.deserializer
# Возможные значения: Java deserializer class.
# Для JSON строк используется StringDeserializer.
# Для Avro/Protobuf используются специализированные deserializers.
value.deserializer=org.apache.kafka.common.serialization.StringDeserializer
```
